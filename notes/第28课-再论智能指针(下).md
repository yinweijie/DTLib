1. 定义`SharedPointer`类，通过计数机制标识内存 (`ref`)：

- 堆内存被指向时：`ref++`
- 指针被置空时：`ref--`
- `ref == 0`时：释放堆内存

类接口为：

```cpp
template <typename T>
class SharedPointer : public Pointer<T>
{
protected:
    int* m_ref;

    void assign(const SharedPointer<T>& rhs);
public:
    SharedPointer(T* p = nullptr);
    SharedPointer(const SharedPointer<T>& rhs);

    SharedPointer<T>& operator=(const SharedPointer<T>& rhs);

    void clear();

    ~SharedPointer();
};
```

当创建第一个智能指针并指向实例对象的时候，同时也会在构造函数里创建一个堆内存，用来记录引用计数。

类定义为：

```cpp
template <typename T>
class SharedPointer : public Pointer<T>
{
protected:
    int* m_ref;

    void assign(const SharedPointer<T>& rhs)
    {
        this->m_pointer = rhs.m_pointer;
        m_ref = rhs.m_ref; // 要保证所有智能指针的计数变量指针都指向同一块计数空间

        if(m_ref)
        {
            (*m_ref)++;
        }
    }
public:
    SharedPointer(T* p = nullptr) : m_ref(nullptr) // 不要忘记初始化成员变量
    {
        if(p)
        {
            m_ref = static_cast<int*>(std::malloc(sizeof(int)));

            if(m_ref)
            {
                *m_ref = 1;
                this->m_pointer = p;
            }
            else
            {
                THROW_EXCEPTION(NoEnoughMemoryException, "No memory to create SharedPointer object ...");
            }
        }
    }
    SharedPointer(const SharedPointer<T>& rhs)
    {
        assign(rhs);
    }

    SharedPointer<T>& operator=(const SharedPointer<T>& rhs)
    {
        if(this != &rhs)
        {
            clear();

            assign(rhs);
        }

        return *this;
    }

    void clear() // 置空指针，并检查引用计数是否为0，为0则清空堆内存
    {
        T* toDel = this->m_pointer;
        int* ref = m_ref;

        this->m_pointer = nullptr;
        m_ref = nullptr;

        if(ref)
        {
            (*ref)--;

            if(*ref == 0)
            {
                free(ref);

                delete toDel;
            }
        }
    }

    ~SharedPointer()
    {
        clear();
    }
};
```
