2. 需要特殊处理的主要步骤有：

- 首元素的插入和删除操作
- 清空和遍历操作
- 插入的序号`i`可以大于`m_length`，用取模操作

类实现为：

```cpp
template <typename T>
class CircleList : public LinkList<T>
{
protected:
    using Node = typename LinkList<T>::Node;

    int mod(int i) const // 返回值范围为0 ~ m_length-1  O(1)
    {
        return (this->m_length == 0) ? 0 : (i % this->m_length);
    }
    Node* last() const // O(n)
    {
        return this->position(this->m_length - 1)->next; // 最末尾的编号为m_length-1，poison(i)的返回值的next才是i的位置
    }

    void last_to_first() const // 把末尾接到第0个元素   // O(n)
    {
        last()->next = this->m_header.next;
    }

public:
    bool insert(const T& e) override // O(n)
    {
        return insert(this->m_length, e);
    }

    bool insert(int i, const T& e) override // O(n)
    {
        bool ret = true;

        i = i % (this->m_length + 1); // O(1)

        ret = LinkList<T>::insert(i, e); // O(n)

        if(ret && i == 0)
        {
            last_to_first(); // 插入到首元素需要额外更新尾节点指向   // O(n)
        }

        return ret;
    }

    bool remove(int i) override // O(n)
    {
        bool ret = true;

        i = mod(i);

        if(i == 0) // 删除0号节点需要特殊处理
        {
            Node* toDel = this->m_header.next;

            if(toDel != nullptr)
            {
                this->m_header.next = toDel->next;
                this->m_length--;

                if(this->m_length != 0)
                {
                    last_to_first(); // O(n)

                    if(this->m_current == toDel)
                    {
                        this->m_current = toDel->next;
                    }
                }
                else // 当链表节点都被删除以后，需要把指针置空
                {
                    this->m_header.next = nullptr;
                    this->m_current = nullptr;
                }

                this->destroy(toDel);
            }
            else // 空链表不能删除节点
            {
                ret = false;
            }
        }
        else
        {
            ret = LinkList<T>::remove(i); // O(n)
        }

        return ret;
    }

    bool set(int i, const T& e) override // O(n)
    {
        return LinkList<T>::set(mod(i), e);
    }

    T get(int i) const override // O(n)
    {
        return LinkList<T>::get(mod(i));
    }

    bool get(int i, T& e) const override // O(n)
    {
        return LinkList<T>::get(mod(i), e);
    }

    int find(const T& e) const override // 遍历过程不能再依靠`next`是否为`nullptr`来判断停止    O(n)
    {
        int ret = -1;
        Node* slider = this->m_header.next;

        for(int i = 0; i < this->m_length; i++)
        {
            if(slider->value == e)
            {
                ret = i;
                break;
            }

            slider = slider->next;
        }

        return ret;
    }

    void clear() override // O(n)
    {
        while(this->m_length > 1) // 删除1号元素可以直接用父类的版本，相对于0号元素来说，不需要多余的更新尾节点的步骤    O(n)
        {
            LinkList<T>::remove(1); // O(1) 每次删除的位置都是编号1，因此时间复杂度为常数
        }

        if(this->m_length == 1) // O(1)
        {
            Node* toDel = this->m_header.next;

            this->m_header.next = nullptr;
            this->m_current = nullptr;
            this->m_length = 0;

            this->destroy(toDel);
        }
    }

    bool move(int i, int step) override // O(n)
    {
        return LinkList<T>::move(mod(i), step);
    }

    bool end() override // 循环链表没有末尾，因此唯一让链表在current()和next()函数中能够停下来的条件就是，链表中没有节点了    O(1)
    {
        return (this->m_length == 0) || (this->m_header.next == nullptr);
    }

    ~CircleList() // O(n)
    {
        clear();
    }
};
```
